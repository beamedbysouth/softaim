local function isTargetVisible(part, Character)
    if not part then return false end
    local origin = Camera.CFrame.Position
    local direction = (part.Position - origin)

    local rayParams = RaycastParams.new()
    rayParams.FilterDescendantsInstances = {LocalPlayer.Character}
    rayParams.FilterType = Enum.RaycastFilterType.Blacklist
    rayParams.IgnoreWater = true

    local result = workspace:Raycast(origin, direction, rayParams)
    return not result or result.Instance:IsDescendantOf(Character)
end

local function isTargetValid(player)
    if not (player and player.Character and player.Character:FindFirstChild("Humanoid") and player.Character.Humanoid.Health > 0) then
        return false
    end

    if Toggles.Wallcheck and Toggles.Wallcheck.Value then
        local hitbox = getTargetHitbox(player.Character)
        if not isTargetVisible(hitbox, player.Character) then
            return false
        end
    end

    return true
end

local function getNearestTarget()
    local mousePos = UserInputService:GetMouseLocation()
    local closestPlayer, shortestDistance = nil, math.huge
    local maxFOV = Options.FOVAmount and Options.FOVAmount.Value or 100

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and isTargetValid(player) then
            local root = player.Character:FindFirstChild("HumanoidRootPart")
            if root then
                local screenPos, onScreen = Camera:WorldToViewportPoint(root.Position)
                if onScreen then
                    local dist = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                    if dist < shortestDistance and dist <= maxFOV then
                        closestPlayer, shortestDistance = player, dist
                    end
                end
            end
        end
    end

    return closestPlayer
end

RunService.RenderStepped:Connect(function(dt)
    -- === World Time Freeze ===
    if freezeTime then
        Lighting.TimeOfDay = string.format("%02d:00:00", freezeHour)
    end

    -- === Custom Crosshair ===
    do
        local visible = crosshairEnabled
        HorizontalLine.Visible = visible
        VerticalLine.Visible = visible

        HorizontalLineLeft.Visible = visible
        HorizontalLineRight.Visible = visible
        VerticalLineUp.Visible = visible
        VerticalLineDown.Visible = visible

        if visible then
            local size = crosshairLength
            local thickness = crosshairWidth
            local center = useMouse and UserInputService:GetMouseLocation() or (Camera.ViewportSize / 2)

            HorizontalLine.Visible = false
            VerticalLine.Visible = false

            -- Update spin angle if enabled and speed > 0
            if spinEnabled and spinSpeed > 0 then
                local radiansPerSecond = spinSpeed * math.pi / 2  -- Adjust speed multiplier as needed
                spinAngle = (spinAngle + radiansPerSecond * dt) % (2 * math.pi)
            else
                spinAngle = 0
            end

            -- Pre-calculate sin and cos for rotation
            local cosA = math.cos(spinAngle)
            local sinA = math.sin(spinAngle)

            -- Function to rotate a point around center
            local function rotatePoint(px, py)
                local dx, dy = px - center.X, py - center.Y
                local rx = dx * cosA - dy * sinA + center.X
                local ry = dx * sinA + dy * cosA + center.Y
                return Vector2.new(rx, ry)
            end

            -- Set properties & positions for each line
            for _, line in pairs({HorizontalLineLeft, HorizontalLineRight, VerticalLineUp, VerticalLineDown}) do
                line.Thickness = thickness
                line.Color = crosshairColor
                line.Visible = true
            end

            -- Calculate rotated positions

            local hlFrom = rotatePoint(center.X - crosshairGap - size, center.Y)
            local hlTo = rotatePoint(center.X - crosshairGap, center.Y)
            HorizontalLineLeft.From = hlFrom
            HorizontalLineLeft.To = hlTo

            local hrFrom = rotatePoint(center.X + crosshairGap, center.Y)
            local hrTo = rotatePoint(center.X + crosshairGap + size, center.Y)
            HorizontalLineRight.From = hrFrom
            HorizontalLineRight.To = hrTo

            local vuFrom = rotatePoint(center.X, center.Y - crosshairGap - size)
            local vuTo = rotatePoint(center.X, center.Y - crosshairGap)
            VerticalLineUp.From = vuFrom
            VerticalLineUp.To = vuTo

            local vdFrom = rotatePoint(center.X, center.Y + crosshairGap)
            local vdTo = rotatePoint(center.X, center.Y + crosshairGap + size)
            VerticalLineDown.From = vdFrom
            VerticalLineDown.To = vdTo

        else
            for _, line in pairs({HorizontalLine, VerticalLine, HorizontalLineLeft, HorizontalLineRight, VerticalLineUp, VerticalLineDown}) do
                line.Visible = false
            end
        end
    end

    -- === FOV Circle ===
    fovCircle.Visible = Toggles.FOVCircle and Toggles.FOVCircle.Value or false
    if fovCircle.Visible then
        local mousePos = UserInputService:GetMouseLocation()
        fovCircle.Position = mousePos
        fovCircle.Radius = Options.FOVAmount and Options.FOVAmount.Value or 100
    end

    -- === Aimbot ===
    if aimbotEnabled then
        if lockedTarget then
            local char = lockedTarget.Character
            local hum = char and char:FindFirstChildOfClass("Humanoid")

            if not char or not hum or hum.Health <= 0 then
                lockedTarget = nil
                aimbotEnabled = false
                if Toggles.AimbotToggle then
                    Toggles.AimbotToggle:SetValue(false)
                end
            end
        end


        if Toggles.Highlight and Toggles.Highlight.Value then
            for player, data in pairs(highlights) do
                if player ~= lockedTarget or not player.Character then
                    if data.lights then for _, light in pairs(data.lights) do light:Destroy() end end
                    if data.highlight then data.highlight:Destroy() end
                    highlights[player] = nil
                end
            end

            if lockedTarget and lockedTarget.Character and not highlights[lockedTarget] then
                local lights = {}
                for _, part in ipairs(lockedTarget.Character:GetDescendants()) do
                    if part:IsA("BasePart") then
                        local light = Instance.new("PointLight")
                        light.Color = Color3.fromRGB(255, 255, 255)
                        light.Range = 4
                        light.Brightness = 1.4
                        light.Shadows = true
                        light.Parent = part
                        table.insert(lights, light)
                    end
                end

                local hl = Instance.new("Highlight")
                hl.FillColor = Color3.fromRGB(255, 255, 255)
                hl.FillTransparency = 0.7
                hl.OutlineColor = Color3.fromRGB(255, 255, 255)
                hl.OutlineTransparency = 0.7
                hl.Adornee = lockedTarget.Character
                hl.Parent = lockedTarget.Character

                highlights[lockedTarget] = {highlight = hl, lights = lights}
            end
        else
            for player, data in pairs(highlights) do
                if data.lights then for _, light in pairs(data.lights) do light:Destroy() end end
                if data.highlight then data.highlight:Destroy() end
            end
            highlights = {}
        end
        if lockedTarget and lockedTarget.Character then
            local char = lockedTarget.Character
            local part = getTargetHitbox(char)

            if part then
                local visible = true
                if Toggles.Wallcheck and Toggles.Wallcheck.Value then
                    visible = isTargetVisible(part, char)
                end

                if visible then
                    local targetPos = part.Position
                    if AutoPredictionEnabled then
                        local hrp = char:FindFirstChild("HumanoidRootPart")
                        if hrp then
                            local pingMs = game:GetService("Stats").Network.ServerStatsItem["Data Ping"]:GetValue()
                            local pingSec = pingMs / 1000
                            local boostFactor = 3 -- tweak if needed

                            targetPos += hrp.Velocity * pingSec * boostFactor
                        end
                    end

                    local camPos = Camera.CFrame.Position
                    local ratio = (mouseOverrideAmount or 50) / 100
                    local lockStrength = (ratio ^ 2) * 0.98

                    Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(camPos, targetPos), lockStrength)
                end
            end
        end
    else
        for player, data in pairs(highlights) do
            if data.lights then for _, light in pairs(data.lights) do light:Destroy() end end
            if data.highlight then data.highlight:Destroy() end
        end
        highlights = {}
    end

    -- === Walkspeed Lock ===
    local humanoid = getHumanoid()
    if humanoid then
        if WalkspeedEnabled and humanoid.WalkSpeed ~= WalkspeedValue then
            humanoid.WalkSpeed = WalkspeedValue
        elseif not WalkspeedEnabled and humanoid.WalkSpeed ~= 16 then
            humanoid.WalkSpeed = 16
        end
    end

    -- === Flight Movement ===
    if FlightEnabled and BodyVelocity and BodyGyro then
        local char = LocalPlayer.Character
        if char then
            local hrp = char:FindFirstChild("HumanoidRootPart")
            if hrp then
                local camCF = Camera.CFrame
                local move = Vector3.zero
                if UserInputService:IsKeyDown(Enum.KeyCode.W) then move += camCF.LookVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.S) then move -= camCF.LookVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.A) then move -= camCF.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.D) then move += camCF.RightVector end
                if UserInputService:IsKeyDown(Enum.KeyCode.Space) then move += Vector3.new(0, 1, 0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then move -= Vector3.new(0, 1, 0) end

                BodyVelocity.Velocity = move.Magnitude > 0 and move.Unit * FlightSpeed or Vector3.zero
                BodyGyro.CFrame = CFrame.new(hrp.Position, hrp.Position + camCF.LookVector)
            end
        end
    end
    -- === Gun Material Logic ===
    do
        local char = LocalPlayer.Character
        if char then
            for _, weaponName in ipairs(weapons) do
                local weapon = char:FindFirstChild(weaponName)
                if weapon and weapon:FindFirstChild("Default") then
                    local default = weapon.Default

                    if gunMaterialEnabled then
                        saveOriginal(default)
                        default.Material = selectedMaterial
                        default.Color = Color3.fromRGB(255, 255, 255)
                        default.TextureID = ""
                    end
                end
            end
        end
    end

    -- === ESP Rendering ===
    if espEnabled then
        local viewportSize = Camera.ViewportSize
        local screenCenter = Vector2.new(viewportSize.X / 2, viewportSize.Y)

        for player, drawings in pairs(ESPStorage) do
            local Character = player.Character
            local humanoid = Character and Character:FindFirstChildOfClass("Humanoid")
            local hrp = Character and Character:FindFirstChild("HumanoidRootPart")

            if Character and humanoid and humanoid.Health > 0 and hrp then
                local box3DSize = Vector3.new(2, 5, 1)
                local halfSize = box3DSize / 2

                local offsets = {
                    Vector3.new(-halfSize.X,  halfSize.Y, -halfSize.Z),
                    Vector3.new( halfSize.X,  halfSize.Y, -halfSize.Z),
                    Vector3.new(-halfSize.X, -halfSize.Y, -halfSize.Z),
                    Vector3.new( halfSize.X, -halfSize.Y, -halfSize.Z),
                    Vector3.new(-halfSize.X,  halfSize.Y,  halfSize.Z),
                    Vector3.new( halfSize.X,  halfSize.Y,  halfSize.Z),
                    Vector3.new(-halfSize.X, -halfSize.Y,  halfSize.Z),
                    Vector3.new( halfSize.X, -halfSize.Y,  halfSize.Z),
                }

                local screenPoints = {}
                local onScreen = false

                for _, offset in ipairs(offsets) do
                    local worldPoint = hrp.Position + offset
                    local screenPoint, visible = worldToScreen(worldPoint)
                    if visible then onScreen = true end
                    screenPoint = Vector2.new(
                        math.clamp(screenPoint.X, 0, viewportSize.X),
                        math.clamp(screenPoint.Y, 0, viewportSize.Y)
                    )
                    table.insert(screenPoints, screenPoint)
                end

                if onScreen then
                    local minX, minY = math.huge, math.huge
                    local maxX, maxY = -math.huge, -math.huge

                    for _, sp in ipairs(screenPoints) do
                        if sp.X < minX then minX = sp.X end
                        if sp.Y < minY then minY = sp.Y end
                        if sp.X > maxX then maxX = sp.X end
                        if sp.Y > maxY then maxY = sp.Y end
                    end

                    local boxPos = Vector2.new(minX, minY)
                    local boxSize = Vector2.new(maxX - minX, maxY - minY)

                    -- Boxes
                    if Settings.Enable2DBoxes then
                        if not drawings.Box then
                            drawings.Box = Drawing.new("Square")
                            drawings.Box.Filled = false
                            drawings.Box.Color = Color3.fromRGB(255, 255, 255)
                            drawings.Box.Thickness = 1
                        end
                        if not drawings.BoxOutline then
                            drawings.BoxOutline = Drawing.new("Square")
                            drawings.BoxOutline.Filled = false
                            drawings.BoxOutline.Color = Color3.new(0, 0, 0)
                            drawings.BoxOutline.Thickness = 1
                        end

                        drawings.Box.Position = boxPos
                        drawings.Box.Size = boxSize
                        drawings.Box.Visible = true

                        drawings.BoxOutline.Position = boxPos - Vector2.new(0.5, 0.5)
                        drawings.BoxOutline.Size = boxSize + Vector2.new(1, 1)
                        drawings.BoxOutline.Visible = true
                    else
                        if drawings.Box then drawings.Box.Visible = false end
                        if drawings.BoxOutline then drawings.BoxOutline.Visible = false end
                    end

                    -- Name ESP on top of the box
                    if Settings.EnableName then
                        if not drawings.NameText then
                            drawings.NameText = Drawing.new("Text")
                            drawings.NameText.Center = true
                            drawings.NameText.Outline = true
                            drawings.NameText.Color = Color3.fromRGB(255, 255, 255)
                            drawings.NameText.Font = 2
                        end
                        local minSize = 10
                        local maxSize = 16
                        local calculatedSize = math.clamp(boxSize.Y / 5, minSize, maxSize)
                        drawings.NameText.Size = calculatedSize

                        drawings.NameText.Text = player.Name
                        -- Position name centered horizontally above the box, 5 pixels above
                        drawings.NameText.Position = Vector2.new(
                            boxPos.X + (boxSize.X / 2),
                            boxPos.Y - 15
                        )
                        drawings.NameText.Visible = true
                    else
                        if drawings.NameText then
                            drawings.NameText.Visible = false
                        end
                    end

                    -- Health bar
                    if Settings.EnableHealth then
                        local healthPercent = math.clamp(humanoid.Health / humanoid.MaxHealth, 0, 1)
                        local barWidth = 2
                        local barHeight = boxSize.Y * healthPercent
                        local barX = boxPos.X - barWidth - 2
                        local barY = boxPos.Y + (boxSize.Y - barHeight)

                        if not drawings.HealthBar then
                            drawings.HealthBar = Drawing.new("Square")
                            drawings.HealthBar.Filled = true
                            drawings.HealthBar.Color = Color3.fromRGB(0, 255, 0)
                        end
                        if not drawings.HealthBarOutline then
                            drawings.HealthBarOutline = Drawing.new("Square")
                            drawings.HealthBarOutline.Filled = false
                            drawings.HealthBarOutline.Color = Color3.new(0, 0, 0)
                            drawings.HealthBarOutline.Thickness = 1
                        end

                        drawings.HealthBar.Position = Vector2.new(barX, barY)
                        drawings.HealthBar.Size = Vector2.new(barWidth, barHeight)
                        drawings.HealthBar.Visible = true

                        drawings.HealthBarOutline.Position = Vector2.new(barX - 0.5, barY - 0.5)
                        drawings.HealthBarOutline.Size = Vector2.new(barWidth + 1, barHeight + 1)
                        drawings.HealthBarOutline.Visible = true
                    else
                        if drawings.HealthBar then drawings.HealthBar.Visible = false end
                        if drawings.HealthBarOutline then drawings.HealthBarOutline.Visible = false end
                    end

                    -- Tracers
                    if tracersEnabled then
                        if not drawings.Tracer then
                            drawings.Tracer = Drawing.new("Line")
                            drawings.Tracer.Color = Color3.fromRGB(255, 255, 255)
                            drawings.Tracer.Thickness = 1
                        end

                        local tracerStart = screenCenter
                        local tracerEnd = boxPos + (boxSize / 2)

                        drawings.Tracer.From = tracerStart
                        drawings.Tracer.To = tracerEnd
                        drawings.Tracer.Visible = true
                    else
                        if drawings.Tracer then drawings.Tracer.Visible = false end
                    end
                else
                    -- Hide all if offscreen
                    if drawings.Box then drawings.Box.Visible = false end
                    if drawings.BoxOutline then drawings.BoxOutline.Visible = false end
                    if drawings.HealthBar then drawings.HealthBar.Visible = false end
                    if drawings.HealthBarOutline then drawings.HealthBarOutline.Visible = false end
                    if drawings.Tracer then drawings.Tracer.Visible = false end
                    if drawings.NameText then drawings.NameText.Visible = false end
                end
            else
                -- Hide all if no valid Character or dead
                if drawings.Box then drawings.Box.Visible = false end
                if drawings.BoxOutline then drawings.BoxOutline.Visible = false end
                if drawings.HealthBar then drawings.HealthBar.Visible = false end
                if drawings.HealthBarOutline then drawings.HealthBarOutline.Visible = false end
                if drawings.Tracer then drawings.Tracer.Visible = false end
                if drawings.NameText then drawings.NameText.Visible = false end
            end
        end
    else
        -- Hide all when ESP disabled
        for _, drawings in pairs(ESPStorage) do
            if drawings.Box then drawings.Box.Visible = false end
            if drawings.BoxOutline then drawings.BoxOutline.Visible = false end
            if drawings.HealthBar then drawings.HealthBar.Visible = false end
            if drawings.HealthBarOutline then drawings.HealthBarOutline.Visible = false end
            if drawings.Tracer then drawings.Tracer.Visible = false end
            if drawings.NameText then drawings.NameText.Visible = false end
        end
    end











    -- === Aspect Ratio Vertical Stretch ===
    if getgenv().res == nil and getgenv().Resolution and getgenv().Resolution[".gg/resolution"] then
        local stretch = getgenv().Resolution[".gg/resolution"]
        Camera.CFrame = Camera.CFrame * CFrame.new(0, 0, 0, 1, 0, 0, 0, stretch, 0, 0, 0, 1)
    end

    if cursorWatermark.Visible then
        local mousePos = UserInputService:GetMouseLocation()
        cursorWatermark.Position = Vector2.new(mousePos.X + 25, mousePos.Y + 10)
    end

    if antiSitEnabled then
        local humanoid = getHumanoid()
        if humanoid and humanoid.Sit then
            humanoid.Sit = false
        end
    end
    -- === Anti-Aim Logic ===
    if EnableAA and hrp and AntiAimType then
        local dt = task.wait() -- delta time

        -- Movement smoothing
        local moveDir = getMoveDirection()
        if moveDir.Magnitude == 0 then
            moveDir = hrp.CFrame.LookVector
        end
        moveDir = Vector3.new(moveDir.X, 0, moveDir.Z)
        if moveDir.Magnitude == 0 then moveDir = Vector3.new(0,0,1) end
        moveDir = moveDir.Unit

        if not lastLookDir then
            lastLookDir = moveDir
        end
        lastLookDir = lastLookDir:Lerp(moveDir, smoothSpeed * dt).Unit

        -- Base CFrame (preserves position)
        local baseCFrame = CFrame.new(hrp.Position, hrp.Position + lastLookDir)

        -- Yaw offset
        local yawOffset = 0

        if AntiAimType == "Sway" then
            swayTime = (swayTime or 0) + dt * AntiAimSpeed
            local swayFrequency = 3 -- cycles/sec at speed=1
            yawOffset = math.sin(swayTime * swayFrequency) * math.rad(swayAngleMax or 45)

        elseif AntiAimType == "Spinbot" then
            spinbotTime = (spinbotTime or 0) + dt * AntiAimSpeed
            yawOffset = math.rad((spinbotTime * 360) % 360)

        elseif AntiAimType == "Jitter" then
            local jitterDelay = 0.05 / AntiAimSpeed -- lower = faster
            if not lastJitterTime or tick() - lastJitterTime >= jitterDelay then
                lastJitterTime = tick()
                currentJitterAngle = math.rad(math.random(0, 360))
            end
            yawOffset = currentJitterAngle
        end

        -- Apply rotation only, preserve position
        hrp.CFrame = CFrame.new(baseCFrame.Position) * CFrame.Angles(0, yawOffset, 0)

    else
        -- Reset state when AA is disabled
        lastLookDir = nil
        swayTime = 0
        spinbotTime = 0
        currentJitterAngle = nil
        lastJitterTime = nil
    end
    if flingEnabled and hrp then
        for _, target in pairs(Players:GetPlayers()) do
            if target ~= player and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                local targetHRP = target.Character.HumanoidRootPart
                if targetHRP.Anchored == false then
                    -- Apply a strong continuous force toward the sky
                    targetHRP.Velocity = Vector3.new(0, 500, 0) -- fling upward
                end
            end
        end
    end
end)

local function RebuildSelection(choice)
    SelectedUserIds = {}

    if type(choice) ~= "table" then return end
    for k, v in pairs(choice) do
        local item = v
        -- Some UI libs pass a map like {["Name"]=true}; handle that too
        if type(k) == "string" and type(v) == "boolean" then
            item = k
        end

        local userId
        if typeof(item) == "Instance" and item:IsA("Player") then
            userId = item.UserId
        elseif type(item) == "string" then
            local plr = Players:FindFirstChild(item)
            if plr then userId = plr.UserId end
        elseif type(item) == "number" then
            userId = item
        end

        if userId then SelectedUserIds[userId] = true end
    end

    -- Debug: print current controllers
    local names = {}
    for _, pl in ipairs(Players:GetPlayers()) do
        if SelectedUserIds[pl.UserId] then table.insert(names, pl.Name) end
    end
    print("[Controllers] Selected -> {" .. table.concat(names, ", ") .. "}")
end

-- === Command handler ===
local function HandleCommand(sender, message)
    -- Ignore self
    if sender == LocalPlayer then
        return
    end

    local msg = string.lower(message)

    if msg == "!bring" then
        local myChar = LocalPlayer.Character
        local theirChar = sender.Character
        local myHRP = myChar and myChar:FindFirstChild("HumanoidRootPart")
        local theirHRP = theirChar and theirChar:FindFirstChild("HumanoidRootPart")

        if myHRP and theirHRP then
            myHRP.CFrame = theirHRP.CFrame + Vector3.new(0, 3, 0)
            print("[Controllers] Teleported to", sender.Name)
        end
    end
end


-- === Chat listeners (new TextChatService first, legacy fallback) ===
local function HookTextChannel(ch)
    if ch and ch:IsA("TextChannel") then
        ch.MessageReceived:Connect(function(message)
            local src = message.TextSource
            if not src then return end
            local sender = Players:GetPlayerByUserId(src.UserId)
            if not sender then return end
            print(("[Chat][%s] %s: %s"):format(ch.Name, sender.Name, message.Text))
            HandleCommand(sender, message.Text)
        end)
    end
end

local function TryHookNewChat()
    local channelsFolder = TextChatService:FindFirstChild("TextChannels")
    if not channelsFolder then return false end

    for _, ch in ipairs(channelsFolder:GetChildren()) do
        HookTextChannel(ch)
    end
    channelsFolder.ChildAdded:Connect(function(ch)
        task.wait()
        HookTextChannel(ch)
    end)
    print("[Controllers] Hooked TextChatService channels")
    return true
end

local function HookLegacyChat()
    local function hookPlr(plr)
        plr.Chatted:Connect(function(msg)
            print(("[Chat][Legacy] %s: %s"):format(plr.Name, msg))
            HandleCommand(plr, msg)
        end)
    end
    for _, p in ipairs(Players:GetPlayers()) do hookPlr(p) end
    Players.PlayerAdded:Connect(hookPlr)
    print("[Controllers] Hooked legacy .Chatted")
end

if not TryHookNewChat() then
    HookLegacyChat()

SelfPlayer:AddCheckbox("ShowArms", {
    Text = "Show arms in first person",
    Default = false,
    Callback = function(val)
        local player = game:GetService("Players").LocalPlayer
        local camera = workspace.CurrentCamera

        local function setArmVisibility(visible)
            local char = player.Character or player.CharacterAdded:Wait()
            for _, partName in ipairs({"Left Arm", "Right Arm", "LeftUpperArm", "RightUpperArm", "LeftLowerArm", "RightLowerArm", "LeftHand", "RightHand"}) do
                local part = char:FindFirstChild(partName, true)
                if part and part:IsA("BasePart") then
                    part.LocalTransparencyModifier = visible and 0 or 1
                end
            end
        end

        -- Monitor camera mode to update based on first-person
        local function onRenderStep()
            if camera.CFrame.Position == camera.Focus.Position then
                setArmVisibility(val) -- in first person
            else
                setArmVisibility(true) -- out of first person: arms visible
            end
        end

        if val then
            game:GetService("RunService"):BindToRenderStep("ShowArmsFirstPerson", Enum.RenderPriority.Camera.Value + 1, onRenderStep)
        else
            game:GetService("RunService"):UnbindFromRenderStep("ShowArmsFirstPerson")
            setArmVisibility(true) -- reset visibility
        end
    end,
})

Fun:AddCheckbox("Fling", {
    Text = "Fling",
    Default = false,
    Callback = function(val)
        flingEnabled = val

        if val then
            -- Set up your character for fling
            for _, part in pairs(player.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CustomPhysicalProperties = PhysicalProperties.new(math.huge, 0.3, 0.5)
                    part.CanCollide = false
                    part.Massless = true
                    part.Velocity = Vector3.new(0,0,0)
                end
            end

            -- Apply BodyAngularVelocity
            local bav = Instance.new("BodyAngularVelocity")
            bav.Name = "FlingBAV"
            bav.Parent = hrp
            bav.AngularVelocity = Vector3.new(0,99999,0)
            bav.MaxTorque = Vector3.new(0, math.huge, 0)
            bav.P = math.huge

            -- Fling loop
            spawn(function()
                while flingEnabled and bav.Parent do
                    bav.AngularVelocity = Vector3.new(0,99999,0)
                    wait(0.2)
                    bav.AngularVelocity = Vector3.new(0,0,0)
                    wait(0.1)
                end
                if bav then bav:Destroy() end
            end)

        else
            -- Reset when fling disabled
            local bav = hrp:FindFirstChild("FlingBAV")
            if bav then bav:Destroy() end
            for _, part in pairs(player.Character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = true
                    part.Massless = false
                    part.CustomPhysicalProperties = PhysicalProperties.new()
                end
            end
        end
    end,
})

local pushLoop -- for disconnecting later

Fun:AddCheckbox("Push", {
    Text = "Push anyone",
    Default = false,
    Callback = function(enabled)
        local localPlayer = PlayersService.LocalPlayer
        local character = localPlayer.Character or localPlayer.CharacterAdded:Wait()
        if not character or not character:FindFirstChild("HumanoidRootPart") then return end

        if enabled then
            -- Start pushing
            pushLoop = RunService.Stepped:Connect(function()
                pcall(function()
                    if not selectedPlayer or not selectedPlayer.Character then return end
                    local targetRoot = selectedPlayer.Character:FindFirstChild("HumanoidRootPart")
                    if not targetRoot then return end

                    -- Push target away from local player
                    local dir = (targetRoot.Position - character.HumanoidRootPart.Position).Unit
                    targetRoot.Velocity = dir * 500 -- adjust force as desired
                end)
            end)
        else
            -- Stop pushing
            if pushLoop then
                pushLoop:Disconnect()
                pushLoop = nil
            end
        end
    end,
})
